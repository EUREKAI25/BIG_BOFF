## IMPLÉMENTATION (Expo Snack)

### preview_service.py

```python
import httpx
import json
from pathlib import Path

class PreviewService:
    SNACK_API = "https://snack.expo.dev/api/v2/snacks"
    
    def __init__(self):
        self.client = httpx.AsyncClient()
    
    async def create_preview(self, project_dir: Path) -> dict:
        """Crée un preview Expo Snack du projet."""
        
        # Lire tous les fichiers du projet
        files = self._read_project_files(project_dir)
        
        # Créer le snack
        snack_data = {
            "name": "Web2App Preview",
            "description": "Generated by Web2App",
            "files": files,
            "dependencies": self._get_dependencies(project_dir),
            "sdkVersion": "50.0.0",
        }
        
        response = await self.client.post(
            self.SNACK_API,
            json=snack_data,
            headers={"Content-Type": "application/json"}
        )
        
        if response.status_code == 200:
            result = response.json()
            return {
                "snack_id": result.get("id"),
                "url": f"https://snack.expo.dev/{result.get('id')}",
                "embed_url": f"https://snack.expo.dev/embedded/{result.get('id')}",
                "qr_code": f"https://qr.expo.dev/snack/{result.get('id')}",
            }
        else:
            return {"error": response.text}
    
    def _read_project_files(self, project_dir: Path) -> dict:
        """Lit tous les fichiers du projet pour Snack."""
        files = {}
        
        # Extensions à inclure
        extensions = [".tsx", ".ts", ".js", ".jsx", ".json"]
        
        for filepath in project_dir.rglob("*"):
            if filepath.is_file() and filepath.suffix in extensions:
                # Path relatif
                rel_path = str(filepath.relative_to(project_dir))
                
                # Ignorer node_modules
                if "node_modules" in rel_path:
                    continue
                
                # Lire le contenu
                with open(filepath, "r", encoding="utf-8") as f:
                    content = f.read()
                
                files[rel_path] = {
                    "type": "CODE",
                    "contents": content,
                }
        
        return files
    
    def _get_dependencies(self, project_dir: Path) -> dict:
        """Extrait les dépendances du package.json."""
        pkg_path = project_dir / "package.json"
        
        if pkg_path.exists():
            with open(pkg_path) as f:
                pkg = json.load(f)
            return pkg.get("dependencies", {})
        
        return {}
    
    def get_embed_html(self, snack_id: str, platform: str = "web") -> str:
        """Génère le HTML pour embed le preview."""
        return f'''
<div data-snack-id="{snack_id}" 
     data-snack-platform="{platform}" 
     data-snack-preview="true" 
     data-snack-theme="light" 
     style="overflow:hidden;background:#f9f9f9;border:1px solid var(--color-border);border-radius:4px;height:505px;width:100%">
</div>
<script async src="https://snack.expo.dev/embed.js"></script>
'''

class ScreenshotService:
```

## API ENDPOINT

### api/preview.py

```python
from fastapi import APIRouter, HTTPException
from ..preview.preview_service import PreviewService
from pathlib import Path

router = APIRouter(prefix="/preview", tags=["preview"])
preview_service = PreviewService()

@router.post("/{project_id}")
async def create_preview(project_id: str):
    """Crée un preview pour un projet généré."""
    project_dir = Path(f"./generated/{project_id}")
    
    if not project_dir.exists():
        raise HTTPException(404, "Project not found")
    
    result = await preview_service.create_preview(project_dir)
    
    if "error" in result:
        raise HTTPException(500, result["error"])
    
    return result

@router.get("/{project_id}/embed")
async def get_embed(project_id: str, platform: str = "web"):
    """Retourne le HTML d'embed."""
    # Récupérer le snack_id depuis la DB ou le recréer
    # Pour simplifier, on assume qu'il existe
    return {
        "html": preview_service.get_embed_html(project_id, platform)
    }
```

## COMPOSANT FRONTEND

### PreviewFrame.tsx

```tsx
import { useState } from 'react';

interface PreviewFrameProps {
  snackId: string;
  platform?: 'ios' | 'android' | 'web';
}

export function PreviewFrame({ snackId, platform = 'web' }: PreviewFrameProps) {
  const [isLoading, setIsLoading] = useState(true);
  
  const platforms = ['ios', 'android', 'web'];
  const [activePlatform, setActivePlatform] = useState(platform);
  
  return (
    <div className="flex flex-col gap-4">
      {/* Platform selector */}
      <div className="flex gap-2">
        {platforms.map((p) => (
          <button
            key={p}
            onClick={() => setActivePlatform(p)}
            className={`px-4 py-2 rounded ${
              activePlatform === p ? 'bg-primary text-white' : 'bg-gray-100'
            }`}
          >
            {p.toUpperCase()}
          </button>
        ))}
      </div>
      
      {/* Preview iframe */}
      <div className="relative aspect-[9/16] max-w-sm mx-auto border rounded-lg overflow-hidden">
        {isLoading && (
          <div className="absolute inset-0 flex items-center justify-center bg-gray-100">
            <span>Chargement...</span>
          </div>
        )}
        
        <iframe
          src={`https://snack.expo.dev/embedded/${snackId}?platform=${activePlatform}&preview=true`}
          className="w-full h-full"
          onLoad={() => setIsLoading(false)}
        />
      </div>
      
      {/* QR Code pour tester sur device */}
      <div className="text-center">
        <p className="text-sm text-gray-600 mb-2">Scanner pour tester sur votre appareil :</p>
        <img 
          src={`https://qr.expo.dev/snack/${snackId}`}
          alt="QR Code"
          className="mx-auto w-32 h-32"
        />
      </div>
    </div>
  );
}
```
