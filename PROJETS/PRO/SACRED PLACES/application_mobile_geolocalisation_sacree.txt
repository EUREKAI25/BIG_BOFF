import pdfplumber
import re
import pandas as pd

# Ouvre le fichier PDF
pdf_path = "/chemin/vers/ton-fichier.pdf"
with pdfplumber.open(pdf_path) as pdf:
    text_content = ""
    
    # Parcours toutes les pages du PDF et extrait le texte
    for page in pdf.pages:
        text_content += page.extract_text()

# Diviser le texte en lignes pour le parcourir plus facilement
lines = text_content.split("\n")

# Filtrer les lignes qui contiennent des lieux (exemple de recherche des noms de lieux et continents)
sections = []
current_section = None
for line in lines:
    if re.match(r'Chapter \d+—(.*)', line):  # Correspond aux sections de chapitre
        if current_section:
            sections.append(current_section)
        current_section = {'title': line, 'places': []}
    elif re.match(r'\d+\.', line):  # Correspond aux lignes qui commencent par des chiffres (liste des lieux)
        if current_section:
            current_section['places'].append(line)

# Ajoute la dernière section si elle n'a pas encore été ajoutée
if current_section:
    sections.append(current_section)

# Organise les données extraites en DataFrame pour faciliter leur utilisation
places_data = []
for section in sections:
    for place in section['places']:
        places_data.append({
            'Section': section['title'],
            'Place': place
        })

# Convertir en DataFrame
df = pd.DataFrame(places_data)

# Affiche un aperçu des données
print(df.head())

# Sauvegarder les données dans un fichier CSV
df.to_csv("extracted_places.csv", index=False)

Regex : Recherche les lignes qui correspondent aux chapitres (par exemple, "Chapter X—") pour identifier les sections, et les lignes commençant par des chiffres pour les lieux.
Structuration : Les lieux extraits sont stockés dans une structure sections qui regroupe les lieux par section (chapitre, continent, etc.).
Pandas : Une fois les données extraites, elles sont organisées dans un tableau et exportées dans un fichier CSV pour une gestion facile des données.

import pdfplumber
import re

pdf_path = "/chemin/vers/ton-fichier.pdf"
with pdfplumber.open(pdf_path) as pdf:
    relevant_pages = []
    
    # Parcours toutes les pages du PDF et cherche les en-têtes de chapitre
    for i, page in enumerate(pdf.pages):

function convertToHeading2() {
  var body = DocumentApp.getActiveDocument().getBody();
  var totalElements = body.getNumChildren();

  // Parcourt tous les éléments du document
  for (var i = 0; i < totalElements; i++) {
    var element = body.getChild(i);
    
    // Vérifie si l'élément est un paragraphe
    if (element.getType() == DocumentApp.ElementType.PARAGRAPH) {
      var paragraph = element.asParagraph();
      var text = paragraph.getText();
      
      // Vérifie si la police est de taille 9 et que le texte commence par un chiffre
      var attributes = paragraph.getAttributes();
      if (attributes[DocumentApp.Attribute.FONT_SIZE] == 9 && /^\d+/.test(text)) {
        // Applique le style "Titre 2" (Heading 2)
        paragraph.setHeading(DocumentApp.ParagraphHeading.HEADING2);
      }
    }
  }

import psycopg2
import gspread
from oauth2client.service_account import ServiceAccountCredentials

# Configuration de l'accès à Google Docs/Sheets via l'API
scope = ["https://spreadsheets.google.com/feeds", "https://www.googleapis.com/auth/drive"]
creds = ServiceAccountCredentials.from_json_keyfile_name("credentials.json", scope)  # Fichier JSON des credentials
client = gspread.authorize(creds)

# Ouvre le fichier Google Docs/Sheets (par exemple en le convertissant d'abord en Google Sheets)
doc = client.open("Nom_du_document").sheet1  # Remplace par le nom réel du document ou de la feuille

# Connexion à la base de données PostgreSQL
conn = psycopg2.connect(
    host="localhost",
    database="nom_de_la_base",
    user="ton_utilisateur",
    password="ton_mot_de_passe"
)
cur = conn.cursor()

# Fonction pour insérer des données dans la table Lieux
def insert_lieu(continent, pays, lieu, description, longitude=None, latitude=None):
    sql = """
    INSERT INTO lieux (continent, pays, lieu, description, long, lat)
    VALUES (%s, %s, %s, %s, %s, %s)
    """
    cur.execute(sql, (continent, pays, lieu, description, longitude, latitude))

# Fonction pour parcourir le document et extraire les sections et les lieux
def process_document():
    rows = doc.get_all_records()  # Si c'est un Google Sheets, ça extrait toutes les lignes

    continent = None
    for row in rows:
        # Vérifie si c'est un titre de Chapitre (Continent)
        if "Chapitre" in row['section']:  # Assure-toi que "section" est la bonne clé
            continent = row['section'].replace("Chapitre", "").strip()
        
        # Sinon, c'est un lieu à traiter
        elif row['lieu']:  # Si c'est un lieu, utilise les informations du lieu
            pays = row.get('pays', '')
            lieu = row['lieu']
            description = row['description']
            # Appelle la fonction d'insertion
            insert_lieu(continent, pays, lieu, description)

    # Sauvegarde les changements dans la base de données
    conn.commit()

# Exécuter le traitement du document
process_document()

# Fermer la connexion à la base de données
cur.close()
conn.close()
